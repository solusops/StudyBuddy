# COGNEE — Distilled Architecture & API Reference
# Source: 60+ examples in cognee/ folder. Zero filler.

================================================================================
1. WHAT IT IS
================================================================================
Open-source AI memory system. Turns raw data → searchable knowledge.
Triple storage: Relational (provenance) + Vector (semantic) + Graph (relationships).
All async Python. Requires LLM_API_KEY in .env (OpenAI default).

================================================================================
2. V1.0 API — 4 CORE OPERATIONS
================================================================================

import cognee
from cognee import SearchType

# --- REMEMBER: ingest data into knowledge graph ---
await cognee.remember(text, dataset_name="ds", self_improvement=False)
# Accepts: str, list[str], file path, S3 URI, URL
# Key params:
#   dataset_name="name"          — group data
#   session_id="sid"             — session-only memory (not in graph until improve())
#   self_improvement=False       — skip auto-improvement pass
#   importance_weight=0.7        — boost node retrieval ranking (0.0-1.0)
#   node_set=["AI","FinTech"]    — tag nodes for filtered retrieval
#   graph_model=MyDataPoint      — custom extraction schema
#   custom_prompt="..."          — override LLM extraction prompt
#   temporal_cognify=True        — extract temporal Event nodes
#   config={"ontology_config":…} — supply RDF/OWL ontology
#   incremental_loading=False    — for URLs
#   preferred_loaders={...}      — e.g. beautiful_soup_loader with extraction_rules

# --- RECALL: query stored memory ---
result = await cognee.recall(query_text="question", datasets=["ds"])
# Key params:
#   query_type=SearchType.GRAPH_COMPLETION  — LLM-synthesized answer from graph
#   query_type=SearchType.CHUNKS            — raw text chunks
#   query_type=SearchType.TEMPORAL          — time-bounded queries
#   datasets=["ds"]                         — scope to datasets
#   session_id="sid"                        — include session context
#   top_k=15                                — max results
#   user=user_obj                           — for multi-tenant

# --- IMPROVE: sync session memory → permanent graph ---
await cognee.improve(dataset="ds", session_ids=["sid"])
# Reads session cache entries, runs add+cognify, persists to graph

# --- FORGET: delete memory ---
await cognee.forget(everything=True)          # nuke all
# Scoped: by item, dataset, or user

3. LEGACY LOW-LEVEL API (still works, used in pipeline internals)
================================================================================
await cognee.add(data, dataset_name="ds")     # ingest only (no cognify)
await cognee.cognify(datasets=["ds"])          # build graph from ingested data
await cognee.search(query_type=..., query_text="...", include_references=True)
await cognee.prune.prune_data()               # delete data
await cognee.prune.prune_system(metadata=True) # delete system metadata

# search() supports include_references=True → appends Evidence section
# SearchType.RAG_COMPLETION → chunk evidence
# SearchType.GRAPH_COMPLETION → entity/graph evidence

6. BUILDING BLOCKS
================================================================================

--- DataPoint: structured data unit → becomes graph node ---
from cognee.infrastructure.engine import DataPoint  # or cognee.low_level
from cognee.infrastructure.engine.models.Edge import Edge

class Person(DataPoint):
    name: str
    knows: List["Person"] = []
    metadata: dict = {"index_fields": ["name"]}  # fields indexed in vector store

# Weighted/typed edges:
bob.knows = (Edge(weight=0.9, relationship_type="friend_of"), charlie)

# Store manually:
from cognee.tasks.storage import add_data_points
await add_data_points([alice, bob, charlie])

--- Task: single processing unit ---
from cognee.modules.pipelines import Task
task = Task(my_async_function, arg1, kwarg1=val)

--- Pipeline: orchestrate tasks ---
tasks = [Task(extract_fn), Task(add_data_points)]
await cognee.run_custom_pipeline(tasks=tasks, data=input_data,
                                  dataset="name", pipeline_name="my_pipe",
                                  user=user)

7. CUSTOM GRAPH MODELS & PROMPTS
================================================================================

# Custom extraction schema — tell cognee what entities to extract:
class Field(DataPoint):
    name: str
    metadata: dict = {"index_fields": ["name"]}

class ProgrammingLanguage(DataPoint):
    name: str
    used_in: List[Field] = None
    metadata: dict = {"index_fields": ["name"]}

await cognee.remember(text, graph_model=ProgrammingLanguage,
                       custom_prompt="Extract programming languages and fields.")

# Custom prompt only (no schema):
await cognee.remember(texts, custom_prompt="Extract only people and cities. "
                      "Connect with 'lives_in'. Ignore all else.")
